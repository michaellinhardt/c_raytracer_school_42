usage : ./RT [(file).rt]

(file).rt :
scene {								; as many as you want
	name { NAME_SCENE }
	camera {pos{x;y;z}, dir{x;y;z}}				; pos = position, dir = direction
	spot {							; as many as you want
		camera {pos{x;y;z}, dir{x;y;z}}				; pos = position, dir = direction
		color { RGB }					; hexa
	}
	ambiance { RGB ; % }					; hexa, intensity, just one
	objects {
		simple_obj {
			name { NAME_OBJ }
			camera {pos{x;y;z}, dir{x;y;z}}
			type { NAME }				; sphere, plan, cone, cylindre, rectangle
			size { a ; b ; c }			; rayon ou longueur ; hauteur ; largeur
			effect { % ; % ; % ; b}			; (b = 1 || 0) order : transparency ; reflexion ; refraction ; negatif
			color { RGB }				; or texture { NAME }
			cut { p{x;y;z}, d{x;y;z} }		; NULL if don't needed 
		}
		complex_obj {
			name { NAME_OBJ }
			camera {pos{x;y;z}, dir{x;y;z}}
			type { NAME }				; pre-define/* a rajouter */ or NULL if you made it with simple obj
			simple_obj {				; as many as needed, without name, NULL if type
			}
		}
	}
}
